  <body>
    <div class="container">
      <h1><i class="fas fa-tasks"></i> Dynamic To-Do List</h1>

      <div class="input-group">
        <input
          type="text"
          id="taskInput"
          class="task-input"
          placeholder="Enter your task..."
          autocomplete="off"
        />
        <button id="addTaskBtn">
          <i class="fas fa-plus"></i>
        </button>
      </div>

      <ul id="tasksList"></ul>

      <div class="stats">
        <p>Total: <span id="totalTasks">0</span></p>
        <p>Completed: <span id="completedTasks">0</span></p>
        <p>Pending: <span id="pendingTasks">0</span></p>
      </div>
    </div>

    <script>
      let tasks = [];

      const taskInput = document.getElementById("taskInput");
      const addTaskBtn = document.getElementById("addTaskBtn");
      const tasksList = document.getElementById("tasksList");

      const totalTasks = document.getElementById("totalTasks");
      const completedTasks = document.getElementById("completedTasks");
      const pendingTasks = document.getElementById("pendingTasks");

      const updateStats = () => {
        const completed = tasks.filter((t) => t.completed).length;
        totalTasks.textContent = tasks.length;
        completedTasks.textContent = completed;
        pendingTasks.textContent = tasks.length - completed;
      };

      const renderTasks = () => {
        tasksList.innerHTML = tasks
          .map(
            (task) => `
            <li data-id="${task.id}" class="${task.completed ? "completed" : ""}">
                <input type="checkbox" ${task.completed ? "checked" : ""}>
                <span>${task.text}</span>
                <button class="delete-btn">
                    <i class="fas fa-trash"></i>
                </button>
            </li>
        `,
          )
          .join("");

        updateStats();
      };

      const addTask = () => {
        const text = taskInput.value.trim();
        if (!text) return;

        tasks.push({
          id: Date.now(),
          text,
          completed: false,
        });

        taskInput.value = "";
        renderTasks();
      };

      tasksList.addEventListener("click", (event) => {
        const li = event.target.closest("li");
        if (!li) return;

        const id = Number(li.dataset.id);

        if (event.target.type === "checkbox") {
          tasks = tasks.map((task) =>
            task.id === id ? { ...task, completed: !task.completed } : task,
          );
        }

        if (event.target.closest(".delete-btn")) {
          tasks = tasks.filter((task) => task.id !== id);
        }

        renderTasks();
      });

      tasksList.addEventListener("dblclick", (event) => {
        if (event.target.tagName !== "SPAN") return;

        const li = event.target.closest("li");
        const id = Number(li.dataset.id);
        const task = tasks.find((t) => t.id === id);

        const newText = prompt("Edit task:", task.text);
        if (newText && newText.trim()) {
          task.text = newText.trim();
          renderTasks();
        }
      });

      addTaskBtn.addEventListener("click", addTask);

      taskInput.addEventListener("keydown", (event) => {
        if (event.key === "Enter") addTask();
      });
    </script>
