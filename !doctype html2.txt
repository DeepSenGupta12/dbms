    <table id="data-table"></table>

    <script>
      const d = [
          { name: "Alice", age: 25, city: "London" },
          { name: "Bob", age: 30, city: "Paris" },
          { name: "Carol", age: 22, city: "New York" },
        ],
        t = document.getElementById("data-table");
      t.innerHTML =
        `<tr>${Object.keys(d[0])
          .map((k) => `<th>${k.toUpperCase()}</th>`)
          .join("")}</tr>` +
        d
          .map(
            (o) =>
              `<tr>${Object.values(o)
                .map((v) => `<td>${v}</td>`)
                .join("")}</tr>`,
          )
          .join("");
    </script>
